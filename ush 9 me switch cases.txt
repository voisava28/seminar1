#include <stdio.h>

int main() {
    int points;
    char grade;

    printf("Enter your numeric grade (0-100): ");
    scanf("%d", &points);

    if (points >= 90 && points <= 100) {
        grade = 'A';
    } else if (points >= 80 && points <= 89) {
        grade = 'B';
    } else if (points >= 70 && points <= 79) {
        grade = 'C';
    } else if (points >= 60 && points <= 69) {
        grade = 'D';
    } else if (points >= 0 && points <= 59) {
        grade = 'F';
    } else {
        printf("Invalid grade\n");
        return 1;
    }

    printf("Your letter grade is: %c\n", grade);

    return 0;
}
